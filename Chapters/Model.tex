\textbf{Parameters:}\\
I : Set of products\\
$I^{1}$: Set of product pairs (i1, i2) that cannot be placed on the same shelf.\\
$I^{2}$: Set of product pairs (i1, i2) that will be included together in the assortment.\\
K : Set of shelves\\
$\pi_i$: Profit made by selling one unit of product i\\
$w_k$ : Width of shelf k\\
$\gamma_k$ : Shelf kâ€™s effect on demand\\
$ds_k$ : Depth of shelf k\\
$dp_i$ : Depth of unit product i\\
$b_i$: Width of a facing for product i\\
$d_i$: Coefficient for demand rate for product i per unit width and one facing\\
$\beta_i$: Space elasticity factor for product i\\
$s^{l}_i$ : Lower bound on the shelf inventory of product, if i is selected in the assortment\\
$s^{u}_i$ : Upper bound on the shelf inventory of product, if i is selected in the assortment\\
\\
\textbf{Notation:}\\
$|I|$: Size of the set of products\\
$|K|$: Size of the set of shelves\\ \\
\textbf{Calculated Data Sets:}\\
$N_{ik}$: number of product i in shelf k per allocation:\\
$N_{ik}=(ds_k/dp_i)-((ds_k/dp_i)\%1)$\\
\\
\textbf{Decision Variables:}\\
$X_{ik}:$ Will the product i be on shelf k. Such that,\\
\[
X_{ik} =
\begin{cases}
1, & if \ i \ is \ allocated \ in \  k \\
0, & otherwise \\
\end{cases}
\]
$f_{ik}$: Number of allocated facings for product i on shelf k.\\
\\
\textbf{Objective function:}\\
\begin{equation*}
\begin{split}
max \sum^{|I|}_{i=1}\sum^{|K|}_{k=1}\pi_i \gamma_k d_i(f_{ik} * b_i)^{\beta_i} \\
\end{split}
\end{equation*}
\textbf{Subjected To:}\\
\begin{equation*}
\begin{split}
\sum^{|K|}_{k=1}X_{ik}\leq 1 \quad \forall i \in I \qquad (Rule \ 1)\\
\end{split}
\end{equation*}
\begin{equation*}
\begin{split}
f_{ik}\leq4 \quad \forall i \in I \quad \forall k \in K  \qquad (Rule \ 2)
\end{split}
\end{equation*}
\begin{equation}\label{constraint3}
\begin{split}
\sum^{|K|}_{k=1}f_{ik}*N_{ik}\leq s^{u}_i*X_{ik} \quad \forall i \in I
\end{split}
\end{equation}
\begin{equation}\label{constraint4}
\begin{split}
\sum^{|K|}_{k=1}f_{ik}*N_{ik}\geq s^{l}_i*X_{ik} \quad \forall i \in I
\end{split}
\end{equation}
Constraint \eqref{constraint3} and \eqref{constraint4} come from Rule 3 and 4 and enforces $f_{ik}$ to be 0 when $X_{ik}$ is 0.\\
\begin{equation*}
\begin{split}
\sum^{|I|}_{i=1}f_{ik}*b_i\leq w_k \quad \forall k \in K (Rule \ 6)
\end{split}
\end{equation*}
\begin{equation*}
\begin{split}
\sum^{|K|}_{k=1}X_{i_{1}k}-\sum^{|K|}_{k=1}X_{i_{2}k}=0 \quad \forall (i_{1},i_{2}) \in I^2  (Rule \ 7) \\ \\
\end{split}
\end{equation*}
\begin{equation*}
\begin{split}
X_{i_{1}k} + X_{i_{2}k} \leq 1 \qquad \forall k \in K \quad  \forall (i_{1},i_{2}) \in I^1  (Rule \ 8)\\
\end{split}
\end{equation*}
